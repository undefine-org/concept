  #+begin_src elixir
  @impl true
  def handle_info(
        %Phoenix.Socket.Broadcast{event: event, payload: notification},
        socket
      )
      when event in ["page_created", "page_updated", "page_archived", "page_restored"] do
    ws = socket.assigns.workspace
    user = socket.assigns.current_user

    socket =
      case Pages.list_tree(actor: user, tenant: ws.id) do
        {:ok, pages} -> assign(socket, :pages, pages)
        _ -> socket
      end

    socket =
      case {event, notification, socket.assigns[:current_page]} do
        {"page_updated", %{data: %{id: id} = page}, %{id: id}} ->
          socket
          |> assign(:current_page, page)
          |> assign(:page_title, page.title)

        {"page_archived", %{data: %{id: id}}, %{id: id}} ->
          socket
          |> assign(:current_page, nil)
          |> assign(:page_title, ws.name)
          |> push_patch(to: ~p"/w/#{ws.slug}")

        _ ->
          socket
      end

    {:noreply, socket}
  end
  #+end_src

- Keep the broader =handle_info(_, socket), do: {:noreply, socket}= catch-all that already exists at the bottom of the module (or add one if missing) so unrelated broadcasts (block_*) routed to this LV are silently ignored.

* Edge Cases
- *Same user editing in two tabs* — the broadcast notification's ~actor.id == current_user.id~. We do *not* filter by self for page broadcasts because the ~PageHeader~ component already updates *its own* ~@page~ assign via ~save_title~; the parent WorkspaceLive needs the broadcast to refresh sidebar/tab title regardless of origin.
- *Renaming a non-current page from the sidebar* — only sidebar refresh applies; ~current_page~ branch falls through to ~_~.
- *Title set to whitespace or only emoji* — the existing ~Pages.rename_page~ accepts any string; the placeholder CSS handles empty via ~:empty~. No additional sanitisation in this fix.
- *Concurrent rename + delete* — if a peer archives the page while we are typing, the ~page_archived~ branch redirects us to the workspace root.
