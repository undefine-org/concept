defmodule Concept.Integration.MCPParityTest do
  use ConceptWeb.ConnCase, async: false
  @moduletag :integration

  describe "MCP parity" do
    test "every exposed tool has a description" do
      tools = AshAi.exposed_tools(otp_app: :concept)
      for tool <- tools do
        refute is_nil(tool.action.description),
          "tool #{tool.name} (#{inspect(tool.resource)}.#{tool.action.name}) missing description"
      end
    end

    test "every public arg on exposed actions has a description" do
      for tool <- AshAi.exposed_tools(otp_app: :concept),
          arg <- tool.action.arguments,
          arg.public? do
        refute is_nil(arg.description),
          "tool #{tool.name} arg #{arg.name} missing description"
      end
    end

    test "every described public action is reachable via tools/list" do
      described =
        for domain <- Application.get_env(:concept, :ash_domains, []),
            resource <- Ash.Domain.Info.resources(domain),
            action <- Ash.Resource.Info.actions(resource),
            action.public? and not is_nil(action.description),
            into: MapSet.new() do
          {resource, action.name}
        end

      exposed =
        AshAi.exposed_tools(otp_app: :concept)
        |> MapSet.new(fn t -> {t.resource, t.action.name} end)

      missing = MapSet.difference(described, exposed)
      assert MapSet.size(missing) == 0,
        "actions described but not exposed: #{inspect(MapSet.to_list(missing))}"
    end

    test "tools/call on a sample read returns same result as code interface", %{conn: conn} do

    end

  end

defmodule Mix.Tasks.Concept.Docs.McpSurface do
  use Mix.Task

  @shortdoc "Generates docs/mcp_surface.md from Ash action metadata."

  @impl Mix.Task
  def run(args) do
    Mix.Task.run("app.start")
    {opts, _} = OptionParser.parse!(args, strict: [check: :boolean])

    output = render_markdown()
    existing = File.read("docs/mcp_surface.md")

    cond do
      opts[:check] and existing == {:ok, output} -> :ok
      opts[:check] -> Mix.raise("docs/mcp_surface.md is out of date; run `mix concept.docs.mcp_surface`")
      true ->
        File.write!("docs/mcp_surface.md", output)
        Mix.shell().info("wrote docs/mcp_surface.md")
    end
  end

* Acceptance

- ~mix test test/integration/mcp_parity_test.exs~ green.
- ~mix concept.docs.mcp_surface~ writes a non-empty file.
- ~mix concept.docs.mcp_surface --check~ exits 0 after running the
  generation; exits non-zero if a description is added without
  regenerating.
- ~mix concept.docs~ runs both surface gen and ExDoc.
