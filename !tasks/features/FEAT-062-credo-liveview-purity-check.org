#+begin_src elixir
defmodule Concept.Credo.Check.LiveViewPurity do
  use Credo.Check,
    id: "EX9001",
    category: :design,
    base_priority: :high,
    explanations: [
      check: """
      LiveView modules must not directly construct Ash queries / changesets
      or call Ecto / the Repo. All data access flows through domain code
      interface functions so that the same call path is available to MCP.

      Forbidden: Ash.Query, Ash.Changeset.for_*, Ecto.Query, Concept.Repo.

      Allowed: Concept.<Domain>.<fn>(...) calls.

      Rationale: see docs/mcp_parity.md.
      """
    ]

  @impl Credo.Check
  def run(source_file, params \\ []) do
    if liveview_module?(source_file) do
      Credo.Code.prewalk(source_file, &traverse(&1, &2, issue_meta(source_file, params)))
    else
      []
    end
  end

  # detect alias/import/use Phoenix.LiveView (or live_view_helpers) in the file
  defp liveview_module?(source_file), do: ...

  defp traverse({:alias, _, [{:__aliases__, _, [:Ash, :Query]} | _]} = ast, issues, meta), do: ...
  # also catch:
  #   Ash.Query.<anything>
  #   Ash.Changeset.for_create / for_update / for_destroy / for_action
  #   Ecto.Query.<anything>
  #   Concept.Repo.<anything>
end
#+end_src

** Scope of files checked

Apply to ~lib/concept_web/live/**/*.ex~ and
~lib/concept_web/components/**/*.ex~. Components shouldn't be doing data

clear in case of a false positive.

** Allow-list for legitimate exceptions

None planned. If a file genuinely needs to escape, it should be moved
into the domain layer. Document this as the resolution path in the
check's explanation text.
